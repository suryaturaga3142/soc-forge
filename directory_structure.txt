OSIC Tools Assisted HDL Developement File Structure

project/                        # Either the whole project is an FPGA one or an ASIC one. No mixing or interchange

    .projectinfo                # Text file for backend parsing, info on project needed for automation like fpga/asic, proj name

    blocks/                     # Holds all project blocks including 1+ project top
        ram/                    # Leaf block (1+ top per block)
            src/                # SystemVerilog RTL (*.sv)
                wrapper.sv      # Top level 1 of ram block
                port.sv         # Top level 2 of ram block
                helper.sv       # Helper module. No core file, no testbench, no vcd, and only optionally accessible to modules within ram block

            inc/                # Headers (*.svh): All should be only optionally accessible to modules within ram block including helper
                ram_pkg.svh     # Package for ram block, may contain params, typedefs, functions, tasks, packages, interfaces, etc

            tb/                 # Testbenches, one per top level module in src. Can see inc/*.svh and any src/*.sv with core file
                wrapper_tb.sv   # Testbench for wrapper.sv
                port_tb.sv      # Testbench for port.sv
                perf_tb.sv      # Testbench not for a specific src/*.sv. Same in all other respects

            waves/              # Waveform gtkw files for gtkwave to access
                wrapper_tb.gtkw # Waveform for wrapper_tb.sv
                port_tb.gtkw    # Waveform for port_tb.sv

            cocotb/             # Optional Python/UVM Cocotb testbenches (not implemented right now, ignore)

            wrapper.core        # FuseSoC configuration for wrapper, ground truth for top and dependencies
            wrapper_tb.core     # FuseSoC config for wrapper testbench
            port.core
            port_tb.core
            perf_tb.core        # Same as other tb cores, just doesn't have an src core in its depend

            .blockinfo          # Info about this block. Now only is it a leaf/global block. Just a text file for backend parsing

        uart/                   # Another leaf block (can instantiate wrapper or port but not helper)
            ...                 # Same structure as ram block

        globals/                # NOT a leaf block. Holds global headers that can be accessed by any *.sv, *_tb.sv, *.svh.
            inc/
                globals_pkg.svh # 1+ global includes for project (can be any number in this dir)

            globals.core        # Not for a single *.svh file. Core file for this entire globals block. Used for access to globals.

            .blockinfo

        globals_alt/            # Another globals block
            ...                 # Same structure as globals block

    build/                      # STRICTLY gitignored. All generated files go here for everything.
        waveforms/              # All iverilog generated *.vcd files go here for gtkwave to access
            *_rtl.vcd
            *_synth.vcd         # *.vcd will be renamed to *_rtl.vcd or *_synth.vcd depending on which it is

        synth_sims/             # Manual synth sims of *.v netlists gen done here inside module dirs labelled by their core names.
            project_ram_wrapper_1.0.0/  # Autogenerated stuff by FuseSoC (don't need to do anything)
            ...                 # For other modules being synthesized and simulated

        ...                     # Other files and dirs generated by FuseSoC and what not

    fusesoc.conf                # FuseSoC config for project, gitignored
    Makefile                    # The top-level orchestrator (may implement user called scripts to call Makefile instead of user)

    .git/
    .gitignore                  # For now only has build and fusesoc.conf
    .gitattributes              # Could be useful in collabs. Force to LF

    constraints/                # Pin Constraints and stuff for FPGA. If project is ASIC, this won't be there
        ice40_generic.pcf
        ice40_generic.core      # FuseSoC config to allow constraints to be accessed path agnostically
        .blockinfo

    # In case of using Sky130 or other tech lib instead of FPGA, may need to remove constraints dir and have something else
    # Or at the very least, the constraints dir will be different.
    # Not implemented yet, will be done after FPGA is working.

${SOME_ARBITRARY_LOCATION}/osic-digital-backend/
    README.md                   # I'll explain what I can here
    LICENSE

    .gitignore
    .gitattributes
    .git/

    templates/
        sv/
            src.sv
            inc.svh
            tb_reg.sv
            tb_single.sv

        core/
            src_fpga.core           # Template for src core file in case of fpga project
            src_asic.core           # Template for src core file in case of asic project
            tb_reg.core             # Template for tb core file w corresponding src module (fpga or asic)
            tb_single.core          # Template for tb core file without a corresponding src module
            inc_global.core         # Template for inc core file for global headers
            constraints_fpga.core   # Template for PCF

        dirinfo/
            blockinfo_leaf          # Leaf blockinfo template
            blockinfo_global        # Global blockinfo template
            blockinfo_constraints   # PCF blockinfo template
            projectinfo_default     # Template for projectinfo

        infra/
            Makefile_default
            gitignore_default
            gitattributes_default

        # Whatever else is needed for a complete project
        # To handle Sky130 projects, may need to have more stuff here. But again, not now.

    scripts/
        # All the different scripts in Bash and Python for Makefiles to make use of,
        # for tools to call for help, for file parsing, anything else needed
        # including creation of new project directories
        # Can also implement a top level bash script called by user that then calls Makefile
        # Will need lots of folders to keep things clean
        prefs/                  # *.env files with general info for bash and python scripts to be able to access

    libs/
        # Custom technology libraries if needed. Whatever OSIC tools comes with should be fine though
        # Generally won't store FPGA stuff like PCFs bc it's not backend and it's small and editable
        # Users can obtain their own PCFs per project.
        # Most likely this directory itself isn't needed. Very low priority since current target is ice40 and Sky130
    
    # Above stuff about covers everything for a complete FPGA project.
    # May need more stuff here for ASIC projects, but will be done after FPGA is working.
